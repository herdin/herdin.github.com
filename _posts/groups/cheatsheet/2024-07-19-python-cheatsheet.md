python3 -m venv {my virtual env name}

# {my virtual env name} 폴더로 가면 bin/activate 실행하거나 command palette 에서 python interpreter 를 {my virtual env name} 로 선택
# 그리고 terminal 을 virutal env 로 실행하여 pip install 하자.
